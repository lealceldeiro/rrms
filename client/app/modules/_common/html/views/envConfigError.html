Error en la configuración del sistema
Debe verificar que exista en el directrio raíz del sistema el archivo env.js con la siguientes estructura: 

(
    function (window) {

        window.__env = window && window.__env ? window.__env : {};

        // API
        window.__env.api = {
        Url:                    'http://localhost:8085/rrmsrest/',

        successFlag:            'success',

        errorMessageFlag:       'errorMessage',

        successMessageFlag:     'successMessage',

        totalCountFlag:         'total',

        itemsFlag:              'items'
        };

        // Base url
        window.__env.baseUrl =      '/';

        window.__env.enableDebug = true;


    }
)(this);
